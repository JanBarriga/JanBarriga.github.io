Piano

En aquest proyecte hem tratat de replicar un piano MIDI amb dfferents botons i tecles, no hem tingut temps per a acabar-ho (degut a una falla del nostre arduino 33 BLE)

LLista de material

  • Arduino nano 33 BLE
  • Cables
  • 8 botons
  • 1 Resistencia 10 kilo ohms
  • 1 altaveu 8 ohms

Aixó es la representació en tinkercad (substituint un arduino nano per un arduino uno, fes click a la imatge per provar ho)


  int PIEZO = 12;
  int BUTTON_C = 11;
  int BUTTON_D = 10;
  int BUTTON_E = 9;
  int BUTTON_F = 8;
  int BUTTON_G = 7;
  int BUTTON_A = 6;
  int BUTTON_B = 8;
  int BUTTON_C2 = 7;
  void setup()
  {
    for(int i = 2; i<=10;i++)
    {
      pinMode(i, INPUT_PULLUP);
    }
    
    pinMode(LED_BUILTIN, OUTPUT);      
  }
  
  
  
  
  
  void loop()
  {
     while(digitalRead(BUTTON_C) == HIGH)
    {
      tone(PIEZO, 262, 100);
      digitalWrite(LED_BUILTIN,HIGH);
    }
    while(digitalRead(BUTTON_D) == HIGH)
    {
      tone(PIEZO, 294, 100);
      digitalWrite(LED_BUILTIN,HIGH);
    }
  
    while(digitalRead(BUTTON_E) == HIGH)
    {
      tone(PIEZO, 330, 100);
      digitalWrite(LED_BUILTIN,HIGH);
    }
    
    while(digitalRead(BUTTON_F) == HIGH)
    {
      tone(PIEZO, 349, 100);
      digitalWrite(LED_BUILTIN,HIGH);
    }
  
    while(digitalRead(BUTTON_G) == HIGH)
    {
      tone(PIEZO, 392, 100);
      digitalWrite(LED_BUILTIN,HIGH);
    }
  
    while(digitalRead(BUTTON_A) == HIGH)
    {
      tone(PIEZO, 440, 100);
      digitalWrite(LED_BUILTIN,HIGH);
    }
  
    while(digitalRead(BUTTON_B) == HIGH)
    {
      tone(PIEZO, 494, 100);
      digitalWrite(LED_BUILTIN,HIGH);
    }      
  
    while(digitalRead(BUTTON_C2) == HIGH)
    {
      tone(PIEZO, 523, 100);
      digitalWrite(LED_BUILTIN,HIGH);
    }  
  
    if(1<2)
    {
      midi();
    }
  
    noTone(PIEZO);
    digitalWrite(LED_BUILTIN,LOW);
  
  }
